@news_blueprint.route('/<int:news_id>')
def detail(news_id):
    news = NewsInfo.query.get(news_id)

    news.click_count += 1
    db.session.commit()

    top_str()

    category_list = NewsCategory.query.filter_by(isDelete=False).order_by(NewsCategory.order)

    click_list = NewsInfo.query \
                     .filter_by(isDelete=False) \
                     .order_by(NewsInfo.click_count.desc())[0:6]

    return render_template(
        'news/detail.html',
        news=news,
        category_list=category_list,
        click_list=click_list,
        title='详细页面'
    )
