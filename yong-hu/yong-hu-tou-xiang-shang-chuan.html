用户头像上传

用户头像上传

  • 加入七牛云的配置，引入七牛云的sdk
  • 在views_user.py中定义视图pic，用于上传用户头像
  • 接收get方式请求，用于展示页面
  • 接收post方式请求，用于接收头像图片
  • 采用ajax方式上传
@user_blueprint.route('/pic', methods=['GET', 'POST'])
def pic():
    user = UserInfo.query.get(session.get('user_id'))
    if request.method == 'GET':
        return render_template('user/user_pic_info.html', user=user)

    avatar = request.files.get('avatar')
    fname = qiniu_upload.upload(avatar.read())

    user.avatar = fname
    db.session.add(user)
    db.session.commit()

    return jsonify(result=1, avatar_url=user.avatar_url)

$('.pic_info').submit(function (e) {
    e.preventDefault();

    $(this).ajaxSubmit({
        url: "/user/pic",
        type: "post",
        dataType: "json",
        success: function (data) {
            if (data.result == 1) {
                $('.now_user_pic').attr('src', data.avatar_url);
                $('.user_center_pic>img',parent.document).attr('src', data.avatar_url);
                $('.lgin_pic',parent.document).attr('src', data.avatar_url);

            } else {
                alert('文件上传失败，请稍候重试');
            }
        }
    });
});
