@user_blueprint.route('/news_release/<int:news_id>', methods=['GET', 'POST'])
def news_release(news_id):
    if news_id > 0:
        news = NewsInfo.query.get(news_id)
    else:
        news = None

    if request.method == 'GET':
        category_list = NewsCategory.query.filter_by(isDelete=False).order_by(NewsCategory.order)
        return render_template(
            'user/user_news_release.html',
            category_list=category_list,
            news=news
        )

    dict1 = request.form
    title = dict1.get('title')
    category = dict1.get('category')
    summary = dict1.get('summary')
    content = dict1.get('content')

    user = g.user
    user.public_count += 1

    if news is None:
        news = NewsInfo()

    news.category_id = int(category)
    news.title = title
    news.summary = summary
    news.context = content
    news.user_id = g.user.id
    news.status = 1

    pic = request.files.get('pic')
    if pic:
        pic_name = qiniu_upload.upload(pic.read())
        news.pic = pic_name

    db.session.add(news)
    db.session.commit()

    return '<script>window.parent.fnChangeMenu(6);window.parent.scrollTo(0, 0);</script>'
