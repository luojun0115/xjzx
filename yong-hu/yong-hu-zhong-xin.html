用户中心

  • 在views_user.py中定义视图index，展示用户中心页面
@user_blueprint.route('/')
def index():
    user_id = session.get('user_id')
    user = UserInfo.query.get(user_id)
    nick_name = user.nick_name
    return render_template('user/user.html',
                           title='用户中心',
                           user=user,
                           nick_name=nick_name)
