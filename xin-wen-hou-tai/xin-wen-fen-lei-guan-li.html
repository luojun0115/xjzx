@admin_blueprint.route('/news_type', methods=['GET', 'POST'])
def news_type():
    if request.method == 'GET':
        category_list = NewsCategory.query.all()
        return render_template(
            'admin/news_type.html',
            category_list=category_list
        )

    dict1 = request.form
    name = dict1.get('name')
    cid = int(dict1.get('cid'))

    if cid == 0:
        category = NewsCategory()
    else:
        category = NewsCategory.query.get(cid)

    category.name = name
    if cid == 0:
        db.session.add(category)
    db.session.commit()

    return jsonify(result=1, cid=category.id)

function getCookie(name) {
    var r = document.cookie.match("\\b" + name + "=([^;]*)\\b");
    return r ? r[1] : undefined;
}

$(function () {
    var $a = $('.edit');
    var $add = $('.addtype');
    var $pop = $('.pop_con');
    var $cancel = $('.cancel');
    var $confirm = $('.confirm');
    var $error = $('.error_tip');
    var $input = $('.input_txt3');
    var sHandler = 'edit';
    var sId = 0;

    $a.click(function () {
        sHandler = 'edit';
        sId = $(this).parent().siblings().eq(0).html();
        $pop.find('h3').html('修改分类');
        $pop.find('.input_txt3').val($(this).parent().prev().html());
        $pop.show();
    });

    $add.click(function () {
        sHandler = 'add';
        $pop.find('h3').html('新增分类');
        $input.val('');
        $pop.show();
    });

    $cancel.click(function () {
        $pop.hide();
        $error.hide();
    });

    $input.click(function () {
        $error.hide();
    });

    $confirm.click(function () {
        var sVal = $input.val();
        if (sVal == '') {
            $error.html('输入框不能为空').show();
            return;
        }
        $pop.hide();
        var cid = 0;
        if (sHandler == 'edit') {
            cid = sId;
        }
        $.post('/admin/news_type', {
            'csrf_token': $('#csrf_token').val(),
            'name': sVal,
            'cid': cid
        }, function (data) {
            if (data.result == 1) {
                location.reload();
            }
        });

    })
})
