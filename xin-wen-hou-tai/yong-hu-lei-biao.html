用户列表

用户列表

  • 定义视图user_list
@admin_blueprint.route('/user_list')
def user_list():
    page = int(request.args.get('page', '1'))

    paginate = UserInfo.query.filter_by(isAdmin=False).order_by(UserInfo.id.desc()).paginate(page, 10, False)
    page_total = paginate.pages

    user_list1 = paginate.items

    return render_template(
        'admin/user_list.html',
        page=page,
        page_total=page_total,
        user_list1=user_list1
    )
  • 调用模板admin/user_list.html
