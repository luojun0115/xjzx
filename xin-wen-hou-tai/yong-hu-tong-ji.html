@admin_blueprint.route('/user_count')
def user_count():
    now = datetime.now()

    if current_app.redis_store.exists('count_total'):
        count_total = int(current_app.redis_store.get('count_total'))
        count_month = int(current_app.redis_store.get('count_month'))
        count_today = int(current_app.redis_store.get('count_today'))
    else:
        month_first = datetime(year=now.year, month=now.month, day=1)
        today_first = datetime(year=now.year, month=now.month, day=now.day)

        count_total = UserInfo.query.filter_by(isAdmin=False).count()
        count_month = UserInfo.query.filter(UserInfo.create_time >= month_first).count()
        count_today = UserInfo.query.filter(UserInfo.create_time >= today_first).count()

        current_app.redis_store.setex('count_total', 3600, count_total)
        current_app.redis_store.setex('count_month', 3600, count_month)
        current_app.redis_store.setex('count_today', 3600, count_today)

    key = '%d-%d-%d_hour' % (now.year, now.month, now.day)
    if current_app.redis_store.exists(key):
        print([bytes.decode(item) for item in current_app.redis_store.hkeys(key)])
        time_list = [bytes.decode(item) for item in current_app.redis_store.hkeys(key)]
        count_list = [int(item) for item in current_app.redis_store.hvals(key)]
    else:
        time_list = []
        count_list = []

    return render_template(
        'admin/user_count.html',
        count_total=count_total,
        count_month=count_month,
        count_today=count_today,
        time_list=time_list,
        count_list=count_list
    )

        user_list = []
        for i in range(1, 1000):
            user = UserInfo()
            user.mobile = str(i)
            user.nick_name=str(i)
            user.password = '123'

        db.session.add_all(user_list)
